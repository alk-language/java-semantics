symbolic $n : int;

sum = 0;
n = $n;
assume $n < 3;

j = 1;
while (j <= n)
invariant sum == (j - 1) * (n * (n + 1) / 2)
{
    i = 1;
    while (i <= n)
    invariant sum == j * (n * (n + 1) / 2) + i * (i - 1) / 2 && i <= n + 1
    {
        sum = sum + i;
        i++;
    }
    j++;
}

assert sum == (j - 1) * (n * (n + 1) / 2);
